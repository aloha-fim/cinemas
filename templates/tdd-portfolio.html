import pytest
from promo_code import validate_promo_code, apply_promo_code

class TestValidatePromoCode:
    """Test promo code validation"""
    
    def test_valid_code_save10(self):
        """SAVE10 is a valid code"""
        result = validate_promo_code("SAVE10")
        assert result["valid"] == True
    
    def test_invalid_code_rejected(self):
        """Unknown code should be rejected"""
        result = validate_promo_code("FAKECODE")
        assert result["valid"] == False
        assert "invalid" in result["error"].lower()
    
    def test_case_insensitive(self):
        """Codes should work regardless of case"""
        result = validate_promo_code("save10")  # lowercase
        assert result["valid"] == True
promo_code.py 
# Step 2: Write minimum code to pass tests

from typing import Dict, Any

# Configuration-driven design
PROMO_CODES: Dict[str, Dict[str, Any]] = {
    "SAVE10": {"discount_percent": 10, "min_tickets": 1},
    "SAVE20": {"discount_percent": 20, "min_tickets": 3},
    "HALF":   {"discount_percent": 50, "min_tickets": 5},
    "STUDENT": {"discount_percent": 15, "requires_id": True},
}

def validate_promo_code(code: str) -> Dict[str, Any]:
    """Validate if a promo code exists."""
    if not code or not code.strip():
        return {"valid": False, "error": "Code cannot be empty"}
    
    code_upper = code.strip().upper()  # Normalize input
    
    if code_upper not in PROMO_CODES:
        return {"valid": False, "error": f"Invalid promo code"}
    
    return {"valid": True, "code": code_upper}

# Step 3: Add comprehensive edge case tests

class TestEdgeCases:
    """Boundary conditions and edge cases"""
    
    def test_whitespace_trimmed(self):
        """Code with whitespace should be trimmed"""
        result = validate_promo_code("  SAVE10  ")
        assert result["valid"] == True
    
    def test_save20_boundary_2_tickets(self):
        """SAVE20 with 2 tickets (below minimum) fails"""
        result = apply_promo_code("SAVE20", num_tickets=2)
        assert result["success"] == False
    
    def test_save20_boundary_3_tickets(self):
        """SAVE20 with 3 tickets (at minimum) passes"""
        result = apply_promo_code("SAVE20", num_tickets=3)
        assert result["success"] == True
    
    def test_student_without_id_fails(self):
        """STUDENT code requires valid student ID"""
        result = apply_promo_code("STUDENT", has_student_id=False)
        assert result["success"] == False
        assert "student" in result["error"].lower()
